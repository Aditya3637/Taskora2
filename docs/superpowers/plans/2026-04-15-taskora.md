**Goal:** Build Taskora — a 60-second decision-making SaaS with War Room, Daily Brief, Initiative rollout, per-entity tracking, analytics, and full subscription/CRM layer.

**Architecture:** Flutter mobile app + Next.js web PWA share a FastAPI backend backed by Supabase (PostgreSQL + Realtime + Auth + Storage). Firebase Cloud Messaging handles push notifications. Razorpay (India) and Stripe (international) handle billing.

**Tech Stack:** Flutter 3, Next.js 14, FastAPI, Supabase, Firebase FCM, Razorpay, Stripe, GA4, Tailwind CSS, Inter/Plus Jakarta Sans fonts.

### Task 5: Row Level Security (RLS)

**Files:**
- Create: `taskora/apps/backend/migrations/004_rls.sql`

- [ ] **Step 1: Write RLS policies**

```sql
-- 004_rls.sql

alter table public.users enable row level security;
alter table public.businesses enable row level security;
alter table public.buildings enable row level security;
alter table public.clients enable row level security;
alter table public.initiatives enable row level security;
alter table public.tasks enable row level security;
alter table public.comments enable row level security;

-- Users: see only themselves
create policy "users_self" on public.users
  for all using (auth.uid() = id);

-- Businesses: members can read their own businesses
create policy "businesses_member_read" on public.businesses
  for select using (
    id in (
      select business_id from public.business_members
      where user_id = auth.uid()
    )
  );

create policy "businesses_owner_write" on public.businesses
  for all using (owner_id = auth.uid());

-- Buildings/Clients: members of the business
create policy "buildings_member" on public.buildings
  for all using (
    business_id in (
      select business_id from public.business_members
      where user_id = auth.uid()
    )
  );

create policy "clients_member" on public.clients
  for all using (
    business_id in (
      select business_id from public.business_members
      where user_id = auth.uid()
    )
  );

-- Initiatives: stakeholders and business members
create policy "initiatives_member" on public.initiatives
  for select using (
    business_id in (
      select business_id from public.business_members
      where user_id = auth.uid()
    )
  );

-- Tasks: primary/secondary stakeholders + business owner
create policy "tasks_stakeholder_read" on public.tasks
  for select using (
    primary_stakeholder_id = auth.uid()
    or id in (
      select task_id from public.task_stakeholders
      where user_id = auth.uid()
    )
    or initiative_id in (
      select id from public.initiatives
      where business_id in (
        select business_id from public.business_members
        where user_id = auth.uid() and role = 'owner'
      )
    )
  );

-- Comments: task stakeholders
create policy "comments_stakeholder" on public.comments
  for all using (
    task_id in (
      select id from public.tasks
      where primary_stakeholder_id = auth.uid()
      union
      select task_id from public.task_stakeholders
      where user_id = auth.uid()
    )
  );
```

```python
# auth.py
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from jose import jwt, JWTError
from config import settings

bearer = HTTPBearer()

def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(bearer)
) -> dict:
    token = credentials.credentials
    try:
        payload = jwt.decode(
            token,
            settings.supabase_jwt_secret,
            algorithms=["HS256"],
            audience="authenticated",
        )
        return {"id": payload["sub"], "email": payload.get("email")}
    except JWTError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid or expired token",
        )
```

```python
from config import settings

    name: str
    type: str  # 'building' | 'client'

@router.post("/")
def create_business(body: BusinessCreate, user=Depends(get_current_user)):
    biz = sb.table("businesses").insert({
        "name": body.name,
        "type": body.type,
        "owner_id": user["id"],
    }).execute().data[0]
    sb.table("business_members").insert({
        "business_id": biz["id"],
        "user_id": user["id"],
        "role": "owner",
    }).execute()
    return biz

@router.get("/")
def list_businesses(user=Depends(get_current_user)):
    memberships = sb.table("business_members").select("business_id").eq("user_id", user["id"]).execute().data
    ids = [m["business_id"] for m in memberships]
    if not ids:
        return []
    return sb.table("businesses").select("*").in_("id", ids).execute().data
```

```python
from fastapi import APIRouter, Depends
from supabase import create_client
from config import settings
from auth import get_current_user
from pydantic import BaseModel
from typing import Optional, List
from datetime import date

router = APIRouter(prefix="/api/v1/initiatives", tags=["initiatives"])
sb = create_client(settings.supabase_url, settings.supabase_service_key)

class EntityAssignment(BaseModel):
    entity_type: str  # 'building' | 'client'
    entity_id: str
    per_entity_end_date: Optional[date] = None

class InitiativeCreate(BaseModel):
    name: str
    description: Optional[str] = None
    business_id: str
    start_date: Optional[date] = None
    target_end_date: Optional[date] = None
    date_mode: str = "uniform"
    entities: List[EntityAssignment] = []

@router.post("/")
def create_initiative(body: InitiativeCreate, user=Depends(get_current_user)):
    init = sb.table("initiatives").insert({
        "name": body.name,
        "description": body.description,
        "business_id": body.business_id,
        "owner_id": user["id"],
        "start_date": str(body.start_date) if body.start_date else None,
        "target_end_date": str(body.target_end_date) if body.target_end_date else None,
        "date_mode": body.date_mode,
    }).execute().data[0]

    if body.entities:
        sb.table("initiative_entities").insert([
            {
                "initiative_id": init["id"],
                "entity_type": e.entity_type,
                "entity_id": e.entity_id,
                "per_entity_end_date": str(e.per_entity_end_date) if e.per_entity_end_date else None,
            }
            for e in body.entities
        ]).execute()

    return init

@router.get("/{initiative_id}")
def get_initiative(initiative_id: str, user=Depends(get_current_user)):
    init = sb.table("initiatives").select("*, initiative_entities(*)").eq("id", initiative_id).single().execute().data
    return init

@router.get("/business/{business_id}")
def list_initiatives(business_id: str, user=Depends(get_current_user)):
    return sb.table("initiatives").select("*, initiative_entities(*)").eq("business_id", business_id).neq("status", "cancelled").execute().data
```

```python
class TaskStatusUpdate(BaseModel):
    status: str
    entity_id: Optional[str] = None  # if None → update global status
    blocker_reason: Optional[str] = None

@router.post("/")
def create_task(body: TaskCreate, user=Depends(get_current_user)):
    task = sb.table("tasks").insert({
        "title": body.title,
        "description": body.description,
        "initiative_id": body.initiative_id,
        "primary_stakeholder_id": body.primary_stakeholder_id,
        "priority": body.priority,
        "due_date": str(body.due_date) if body.due_date else None,
        "date_mode": body.date_mode,
        "entity_inheritance": body.entity_inheritance,
    }).execute().data[0]

    # Assign primary stakeholder
    sb.table("task_stakeholders").insert({
        "task_id": task["id"],
        "user_id": body.primary_stakeholder_id,
        "role": "primary",
    }).execute()

    # Entities
    if body.entities:
        sb.table("task_entities").insert([
            {
                "task_id": task["id"],
                "entity_type": e.entity_type,
                "entity_id": e.entity_id,
                "per_entity_end_date": str(e.per_entity_end_date) if e.per_entity_end_date else None,
            }
            for e in body.entities
        ]).execute()

    return task

@router.patch("/{task_id}/status")
def update_task_status(task_id: str, body: TaskStatusUpdate, user=Depends(get_current_user)):
    now = datetime.utcnow().isoformat()
    if body.entity_id:
        # Per-entity status update
        sb.table("task_entities").update({
            "per_entity_status": body.status,
            "last_updated": now,
        }).eq("task_id", task_id).eq("entity_id", body.entity_id).execute()
    else:
        # Global status update
        update = {"status": body.status, "last_updated": now}
        if body.blocker_reason:
            update["blocker_reason"] = body.blocker_reason
        sb.table("tasks").update(update).eq("id", task_id).execute()
    return {"ok": True}

@router.get("/{task_id}")
def get_task(task_id: str, user=Depends(get_current_user)):
    return sb.table("tasks").select(
        "*, task_entities(*), task_stakeholders(*), subtasks(*), comments(*), attachments(*)"
    ).eq("id", task_id).single().execute().data
```

```python
from fastapi import APIRouter, Depends
from supabase import create_client

@router.get("/")
def get_daily_brief(user=Depends(get_current_user)):
    uid = user["id"]
    today = date.today()
    week_out = (today + timedelta(days=7)).isoformat()
    today_str = today.isoformat()

    # 1. Decisions pending (Primary, status=pending_decision)
    decisions = sb.table("tasks").select("*, task_entities(*), initiative_id").eq(
        "primary_stakeholder_id", uid
    ).eq("status", "pending_decision").order("created_at").execute().data

    # 2. Overdue tasks (primary or secondary)
    secondary_task_ids = [
        r["task_id"] for r in
        sb.table("task_stakeholders").select("task_id").eq("user_id", uid).execute().data
    ]
    overdue = sb.table("tasks").select("*, task_entities(*)").lt(
        "due_date", today_str
    ).neq("status", "done").neq("status", "archived").in_(
        "id", secondary_task_ids
    ).execute().data if secondary_task_ids else []

    # 3. Stale tasks (last_updated > 7 days ago, user is stakeholder)
    stale_threshold = (today - timedelta(days=7)).isoformat()
    stale = sb.table("tasks").select("*, task_entities(*)").lt(
        "last_updated", stale_threshold
    ).neq("status", "done").neq("status", "archived").in_(
        "id", secondary_task_ids
    ).execute().data if secondary_task_ids else []

    # 4. Due this week
    due_this_week = sb.table("tasks").select("*, task_entities(*)").gte(
        "due_date", today_str
    ).lte("due_date", week_out).in_(
        "id", secondary_task_ids
    ).execute().data if secondary_task_ids else []

    # 5. Blocked tasks
    blocked = sb.table("tasks").select("*, task_entities(*)").eq(
        "status", "blocked"
    ).in_("id", secondary_task_ids).execute().data if secondary_task_ids else []

    return {
        "user_id": uid,
        "generated_at": today_str,
        "decisions_pending": decisions,
        "overdue": overdue,
        "stale": stale,
        "due_this_week": due_this_week,
        "blocked": blocked,
    }
```

```python
from fastapi import APIRouter, Depends
from supabase import create_client
from config import settings
from auth import get_current_user
from notifications import send_push_to_user
from pydantic import BaseModel
from typing import Optional, List
from datetime import datetime

router = APIRouter(prefix="/api/v1/tasks/{task_id}/decisions", tags=["decisions"])
sb = create_client(settings.supabase_url, settings.supabase_service_key)

class DecisionAction(BaseModel):
    action: str  # approve|reject|delegate|request_info|escalate|snooze
    reason: Optional[str] = None
    entity_ids: Optional[List[str]] = None  # None = all entities
    delegate_to: Optional[str] = None  # user_id for delegate
    snooze_hours: Optional[int] = None

@router.post("/")
def take_decision(task_id: str, body: DecisionAction, user=Depends(get_current_user)):
    now = datetime.utcnow().isoformat()
    task = sb.table("tasks").select("*, task_stakeholders(*)").eq("id", task_id).single().execute().data

    if body.action == "approve":
        if body.entity_ids:
            for eid in body.entity_ids:
                sb.table("task_entities").update({"per_entity_status": "done", "last_updated": now}).eq("task_id", task_id).eq("entity_id", eid).execute()
        else:
            sb.table("tasks").update({"status": "done", "last_updated": now}).eq("id", task_id).execute()

    elif body.action == "reject":
        if not body.reason:
            return {"error": "reason required for reject"}
        sb.table("tasks").update({"status": "in_progress", "last_updated": now}).eq("id", task_id).execute()

    elif body.action == "delegate":
        sb.table("tasks").update({"primary_stakeholder_id": body.delegate_to, "last_updated": now}).eq("id", task_id).execute()
        send_push_to_user(body.delegate_to, "Task Delegated to You", task["title"], {"task_id": task_id})

    elif body.action == "escalate":
        sb.table("tasks").update({"priority": "urgent", "last_updated": now}).eq("id", task_id).execute()
        for s in task.get("task_stakeholders", []):
            send_push_to_user(s["user_id"], "Task Escalated", task["title"], {"task_id": task_id})

    # Log decision
    sb.table("decision_log").insert({
        "task_id": task_id,
        "user_id": user["id"],
        "action": body.action,
        "reason": body.reason,
        "entity_ids_affected": body.entity_ids or [],
        "created_at": now,
    }).execute()

    return {"ok": True}
```

```dart
import 'dart:convert';
import 'package:http/http.dart' as http;
import 'package:supabase_flutter/supabase_flutter.dart';

class ApiService {
  static const _base = String.fromEnvironment('API_BASE_URL', defaultValue: 'http://localhost:8000');

  static Future<Map<String, dynamic>> get(String path) async {
    final token = Supabase.instance.client.auth.currentSession?.accessToken;
    final res = await http.get(
      Uri.parse('$_base$path'),
      headers: {'Authorization': 'Bearer $token', 'Content-Type': 'application/json'},
    );
    return jsonDecode(res.body) as Map<String, dynamic>;
  }

  static Future<Map<String, dynamic>> post(String path, Map<String, dynamic> body) async {
    final token = Supabase.instance.client.auth.currentSession?.accessToken;
    final res = await http.post(
      Uri.parse('$_base$path'),
      headers: {'Authorization': 'Bearer $token', 'Content-Type': 'application/json'},
      body: jsonEncode(body),
    );
    return jsonDecode(res.body) as Map<String, dynamic>;
  }

  static Future<Map<String, dynamic>> getDailyBrief() => get('/api/v1/daily-brief/');
}
```

```python
from fastapi import APIRouter, Depends, Request, HTTPException
from supabase import create_client
from config import settings
from auth import get_current_user
import razorpay
import stripe
from pydantic import BaseModel
from datetime import datetime, timedelta

router = APIRouter(prefix="/api/v1/billing", tags=["billing"])
sb = create_client(settings.supabase_url, settings.supabase_service_key)
rz = razorpay.Client(auth=(settings.razorpay_key_id, settings.razorpay_key_secret))
stripe.api_key = settings.stripe_secret_key

PLAN_AMOUNT_INR = {"pro": 99900, "business": 299900}  # paise
RAZORPAY_PLAN_IDS = {"pro": "plan_pro_monthly", "business": "plan_biz_monthly"}  # set in Razorpay dashboard

class CreateSubscription(BaseModel):
    business_id: str
    plan: str  # 'pro' | 'business'
    billing_cycle: str = "monthly"  # 'monthly' | 'annual'
    currency: str = "INR"

@router.post("/subscribe")
def create_subscription(body: CreateSubscription, user=Depends(get_current_user)):
    if body.currency == "INR":
        sub = rz.subscription.create({
            "plan_id": RAZORPAY_PLAN_IDS[body.plan],
            "customer_notify": 1,
            "quantity": 1,
            "total_count": 12 if body.billing_cycle == "annual" else 120,
        })
        sb.table("subscriptions").upsert({
            "business_id": body.business_id,
            "plan": body.plan,
            "status": "trialing",
            "billing_cycle": body.billing_cycle,
            "razorpay_subscription_id": sub["id"],
            "trial_start": datetime.utcnow().isoformat(),
            "trial_end": (datetime.utcnow() + timedelta(days=60)).isoformat(),
        }).execute()
        return {"razorpay_subscription_id": sub["id"], "razorpay_key": settings.razorpay_key_id}
    else:
        # Stripe for international
        customer = stripe.Customer.create(email=user["email"])
        price_id = f"price_stripe_{body.plan}_{body.billing_cycle}"
        sub = stripe.Subscription.create(
            customer=customer.id,
            items=[{"price": price_id}],
            trial_period_days=60,
        )
        sb.table("subscriptions").upsert({
            "business_id": body.business_id,
            "plan": body.plan,
            "status": "trialing",
            "billing_cycle": body.billing_cycle,
            "stripe_subscription_id": sub.id,
            "trial_start": datetime.utcnow().isoformat(),
            "trial_end": (datetime.utcnow() + timedelta(days=60)).isoformat(),
        }).execute()
        return {"stripe_client_secret": sub.latest_invoice.payment_intent.client_secret}

@router.post("/razorpay-webhook")
async def razorpay_webhook(request: Request):
    payload = await request.body()
    signature = request.headers.get("X-Razorpay-Signature", "")
    try:
        rz.utility.verify_webhook_signature(payload.decode(), signature, settings.razorpay_key_secret)
    except Exception:
        raise HTTPException(status_code=400, detail="Invalid signature")

    event = await request.json()
    if event["event"] == "subscription.activated":
        sub_id = event["payload"]["subscription"]["entity"]["id"]
        sb.table("subscriptions").update({"status": "active"}).eq("razorpay_subscription_id", sub_id).execute()
    elif event["event"] == "subscription.cancelled":
        sub_id = event["payload"]["subscription"]["entity"]["id"]
        sb.table("subscriptions").update({"status": "cancelled"}).eq("razorpay_subscription_id", sub_id).execute()
    return {"ok": True}

@router.get("/status/{business_id}")
def get_subscription_status(business_id: str, user=Depends(get_current_user)):
    sub = sb.table("subscriptions").select("*").eq("business_id", business_id).order("created_at", desc=True).limit(1).execute().data
    if not sub:
        return {"plan": "free", "status": "none"}
    return sub[0]
```

```python
    since = (datetime.now(timezone.utc) - timedelta(days=days)).isoformat()

    done_tasks = sb.table("tasks").select("created_at, last_updated").eq(
        "primary_stakeholder_id", uid
    ).eq("status", "done").gte("last_updated", since).execute().data

    tat_values = []
    for t in done_tasks:
        created = datetime.fromisoformat(t["created_at"])
        updated = datetime.fromisoformat(t["last_updated"])
        tat_values.append((updated - created).total_seconds() / 3600)

    avg_tat_hours = sum(tat_values) / len(tat_values) if tat_values else 0

    decisions = sb.table("decision_log").select("created_at").eq(
        "user_id", uid
    ).gte("created_at", since).execute().data

    overdue = sb.table("tasks").select("id", count="exact").eq(
        "primary_stakeholder_id", uid
    ).lt("due_date", datetime.now(timezone.utc).date().isoformat()).neq("status", "done").execute().count

    return {
        "avg_tat_hours": round(avg_tat_hours, 1),
        "tasks_completed": len(done_tasks),
        "decisions_made": len(decisions),
        "overdue_count": overdue,
        "timeframe_days": days,
    }

@router.get("/initiative/{initiative_id}")
def initiative_analytics(initiative_id: str, user=Depends(get_current_user)):
    tasks = sb.table("tasks").select("*, task_entities(*)").eq(
        "initiative_id", initiative_id
    ).execute().data

    total = len(tasks)
    done = len([t for t in tasks if t["status"] == "done"])
    blocked = len([t for t in tasks if t["status"] == "blocked"])

    entity_progress: dict = {}
    for t in tasks:
        for e in t.get("task_entities", []):
            eid = e["entity_id"]
            if eid not in entity_progress:
                entity_progress[eid] = {"total": 0, "done": 0}
            entity_progress[eid]["total"] += 1
            if e["per_entity_status"] == "done":
                entity_progress[eid]["done"] += 1

    return {
        "total_tasks": total,
        "done": done,
        "blocked": blocked,
        "completion_pct": round(done / total * 100) if total else 0,
        "entity_progress": entity_progress,
    }
```
